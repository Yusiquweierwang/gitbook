## SQL语句
```SQL
CREATE DATABASE `NAME`;
//创建数据库

DROP DATABASE `NAME`;
//删除数据库

SHOW DATABASE;
//查看已创建的数据库

USE  `database`;
//使用数据库

CREATE TABLE `table`
//创建表格

DESCRIBE `database`;
//查看表格属性

ALTER TABLE  `table` ADD `property` INT;
//为表格添加属性

ALTER TABLE  `table` DROP `property` INT;
//删除表格属性

INSERT INTO `student` VALUES()
//填入资料

```
